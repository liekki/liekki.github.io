package pakkaaminen;

import java.io.Serializable;

/**
 * Binary tree to be used with Huffman coding
 *
 * @author liekki
 */
public class HuffmanTree implements Serializable {

    private char value;
    private int weight;
    private HuffmanTree left;
    private HuffmanTree right;

    public HuffmanTree() {
        this.value = Character.MIN_VALUE;
        this.weight = Integer.MIN_VALUE;
        this.left = null;
        this.right = null;
    }

    public HuffmanTree(char value) {
        this.value = value;
        this.weight = 0;
        this.left = null;
        this.right = null;
    }

    public HuffmanTree(char value, int weight) {
        this.value = value;
        this.weight = weight;
        this.left = null;
        this.right = null;
    }

    public HuffmanTree(int weight, HuffmanTree left, HuffmanTree right) {
        this.value = Character.MIN_VALUE;
        this.weight = weight;
        this.left = left;
        this.right = right;
    }

    public int getWeight() {
        return this.weight;
    }

    public void setValue(char value) {
        this.value = value;
    }

    public char getValue() {
        return this.value;
    }

    public void setLeft(HuffmanTree left) {
        this.left = left;
    }

    public void setRight(HuffmanTree right) {
        this.right = right;
    }

    public HuffmanTree getLeft() {
        return this.left;
    }

    public HuffmanTree getRight() {
        return this.right;
    }

    public void debug() {
        System.out.println(this.weight);

        if (this.left != null) {
            System.out.print("L: ");
            this.left.debug();
        }
        if (this.right != null) {
            System.out.print("R: ");
            this.right.debug();

        }
    }
}
