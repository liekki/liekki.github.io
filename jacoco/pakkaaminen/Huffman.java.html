package pakkaaminen;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.util.HashMap;
import java.util.Map;

/**
 *
 * @author liekki
 */
public final class Huffman {

    private String input;
    private HashMap<Character, Integer> frequencyTable;
    private MinHeap heap;
    private HashMap<Character, Integer> codes;
    private HuffmanTree tree;
    
    public Huffman(String input) {
        this.input = input;
        this.codes = new HashMap<>();

        this.frequencyTable = generateFrequencyTable(this.input);
        this.heap = generateHuffmanHeap(this.frequencyTable, 256);

        this.tree = generateHuffmanTree(heap);
      
        generateCodes(this.tree, this.codes, "");

        String encoded = "";
        try {
            encoded = serializeHuffmanTree(this.tree) + "#" + encode(this.codes, this.input);

        } catch(IOException e) {
            System.out.println("Exception: " + e.getMessage());
            return;
        }
        
        
        System.out.println(this.input);
        System.out.println("Raw input size: " + input.length() * 8 + " bits.");
        System.out.println(encoded);
        System.out.println("Encoded size: " + encoded.length() + " bits.");
    }

    
    /**
     * Maps each symbol in the input with the corresponding encoding from the encoding table
     * 
     * @param codes Huffman encoding table
     * @param input string to be encoded
     * @return 
     */
    public String encode(HashMap codes, String input) {
        StringBuilder output = new StringBuilder();

        for (int i = 0; i < input.length(); i++) {
            output.append(codes.get(input.charAt(i)));
        }

        return output.toString();
    }

    /**
     * Generates the final Huffman tree by picking two nodes with the smallest
     * weights and adding them back into the heap until the size of the heap is
     * 1.
     *
     * @param heap contains all the nodes as HuffmanTree objects of size 1
     * @return final Huffman tree
     */
    public HuffmanTree generateHuffmanTree(MinHeap heap) {
        while (heap.getSize() > 1) {
            HuffmanTree right = heap.pop();
            HuffmanTree left = heap.pop();
            HuffmanTree current = new HuffmanTree(left.getWeight() + right.getWeight(), left, right);
            heap.insert(current);
        }

        HuffmanTree tree = heap.pop();

        return tree;
    }

    /**
     * Generates a frequency table containing the number of occurrences for each 
     * individual symbol in the input.
     * 
     * @param str input string 
     * @return the resulting frequency table as a HashMap
     */
    public static HashMap generateFrequencyTable(String str) {
        HashMap<Character, Integer> table = new HashMap<>();

        for (int i = 0; i < str.length(); i++) {
            if (table.containsKey(str.charAt(i))) {
                Integer previous = table.get(str.charAt(i));
                table.put(str.charAt(i), previous + 1);
            } else {
                table.put(str.charAt(i), 1);
            }
        }

        return table;
    }

    /**
     * Initializes the heap by inserting 1-sized HuffmanTrees into a heap.
     * 
     * @param table frequency table
     * @param size size of the heap
     * @return 
     */
    public MinHeap generateHuffmanHeap(HashMap<Character, Integer> table, int size) {
        MinHeap heap = new MinHeap(size);
        for (Map.Entry<Character, Integer> entry : table.entrySet()) {
            Character symbol = entry.getKey();
            Integer weight = entry.getValue();

            HuffmanTree node = new HuffmanTree(symbol, weight);
            heap.insert(node);
        }

        return heap;
    }

    /**
     * Recursive helper that generates the encoding table for each individual symbol in the string and
     * stores them in a HashMap
     * 
     * @param tree
     * @param codes
     * @param code 
     */
    public static void generateCodes(HuffmanTree tree, HashMap codes, String code) {
        if (tree.getLeft() == null && tree.getRight() == null) {
            System.out.println(tree.getValue() + ": " + code);
            codes.put(tree.getValue(), code);
            return;
        }

        generateCodes(tree.getLeft(), codes, code + "0");
        generateCodes(tree.getRight(), codes, code + "1");
    }
    
    
    /**
     * returns a serialized string representation of the Huffman tree
     * @param tree
     * @return
     * @throws IOException 
     */
    public String serializeHuffmanTree(HuffmanTree tree) throws IOException {
        ByteArrayOutputStream bo = new ByteArrayOutputStream();
        ObjectOutputStream so = new ObjectOutputStream(bo);
        so.writeObject(tree);
        so.flush();
        return bo.toString();
    }
    
    
    /**
     * De-serializes the string representation of a Huffman tree back to object form
     * 
     * @param serialized
     * @return
     * @throws IOException 
     */
    public HuffmanTree deserializeHuffmanTree(String serialized) throws IOException {
        byte b[] = serialized.getBytes(); 
        ByteArrayInputStream bi = new ByteArrayInputStream(b);
        ObjectInputStream si = new ObjectInputStream(bi);
        
        HuffmanTree output = new HuffmanTree();
        try {
            output = (HuffmanTree) si.readObject();   
        } catch (Exception e) {
            System.out.println(e.getMessage());
        }
        
        return output;
    }
}
