package pakkaaminen;

import java.io.BufferedInputStream;
import java.io.ByteArrayOutputStream;
import java.io.FileInputStream;
import java.io.FileOutputStream;

/**
 * File utility class providing operations such as read/write
 * @author liekki
 */
public class File {

    /**
     * Writes provided bytes to a file
     *
     * @param filename filename to be written to
     * @param bytes bytes to write
     */
    public static void write(String filename, byte[] bytes) throws java.io.IOException {
        try ( FileOutputStream stream = new FileOutputStream("./" + filename)) {
            stream.write(bytes);
        }
    }

    /**
     * Reads file from disk and returns the contents as byte array
     *
     * @param filename file to be read
     * @return bytes of given filename
     * @throws java.io.IOException
     * @throws java.io.FileNotFoundException
     */
    public static byte[] read(String filename) throws java.io.IOException, java.io.FileNotFoundException {
        FileInputStream fileInput = new FileInputStream(filename);
        BufferedInputStream stream = new BufferedInputStream(fileInput);
        ByteArrayOutputStream buffer = new ByteArrayOutputStream();

        int result = stream.read();
        while (result != -1) {
            buffer.write((byte) result);
            result = stream.read();
        }

        byte[] bytes = buffer.toByteArray();
        
        return bytes;
    }
}
