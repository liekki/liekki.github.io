package pakkaaminen;

/**
 * Exercise project for the Helsinki University course "Data Structures and
 * Algorithms". Testing and demonstration of different compression algorithms.
 */
public class Main {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        Huffman huffman = new Huffman();

        try {
            huffman.encodeString("this is an example of a huffman tree");
            huffman.write("test.huff");
        } catch (Exception e) {
            System.out.println("Exception while encoding/writing: " + e.getMessage());
        }

        try {
            System.out.println(huffman.read("test.huff"));
        } catch (Exception e) {
            System.out.println("Exception while decoding/reading: " + e.getMessage());
        }
    }

}
